4. Netbeans compilation commands:

# Compile Function.cpp
mkdir -p build/Debug/GNU-Linux-x86; rm -rf "build/Debug/GNU-Linux-x86/*.o.d"
g++    -c -g -DUSE_LIBS -I../../SuiteSparse/CHOLMOD/Include -I/home/chung/Documents/MATLAB/SuiteSparse/SuiteSparse_config \ 
  -MMD -MP -MF "build/Debug/GNU-Linux-x86/Function.o.d" -o build/Debug/GNU-Linux-x86/Function.o Function.cpp

# Compile Matrix.cpp
g++    -c -g -DUSE_LIBS -I../../SuiteSparse/CHOLMOD/Include -I/home/chung/Documents/MATLAB/SuiteSparse/SuiteSparse_config \
  -MMD -MP -MF "build/Debug/GNU-Linux-x86/Matrix.o.d" -o build/Debug/GNU-Linux-x86/Matrix.o Matrix.cpp

# Compile MatrixFactory.cpp
g++    -c -g -DUSE_LIBS -I../../SuiteSparse/CHOLMOD/Include -I/home/chung/Documents/MATLAB/SuiteSparse/SuiteSparse_config \
  -MMD -MP -MF "build/Debug/GNU-Linux-x86/MatrixFactory.o.d" -o build/Debug/GNU-Linux-x86/MatrixFactory.o MatrixFactory.cpp

# Compile Quadratic.cpp
g++    -c -g -DUSE_LIBS -I../../SuiteSparse/CHOLMOD/Include -I/home/chung/Documents/MATLAB/SuiteSparse/SuiteSparse_config \
  -MMD -MP -MF "build/Debug/GNU-Linux-x86/Quadratic.o.d" -o build/Debug/GNU-Linux-x86/Quadratic.o Quadratic.cpp

# Compile main
g++    -c -g -DUSE_LIBS -I../../SuiteSparse/CHOLMOD/Include -I/home/chung/Documents/MATLAB/SuiteSparse/SuiteSparse_config \
  -MMD -MP -MF "build/Debug/GNU-Linux-x86/main.o.d" -o build/Debug/GNU-Linux-x86/main.o main.cpp

# Link
g++     -o dist/Debug/GNU-Linux-x86/forbesc build/Debug/GNU-Linux-x86/Function.o \
  build/Debug/GNU-Linux-x86/Matrix.o build/Debug/GNU-Linux-x86/MatrixFactory.o \
  build/Debug/GNU-Linux-x86/Quadratic.o build/Debug/GNU-Linux-x86/main.o \
  -L/home/chung/Documents/MATLAB/SuiteSparse/CHOLMOD/Lib \
  -L/home/chung/Documents/MATLAB/SuiteSparse/AMD/Lib \
  -L/home/chung/Documents/MATLAB/SuiteSparse/COLAMD/Lib \
  -L/home/chung/Documents/MATLAB/SuiteSparse/SuiteSparse_config \
  -L/home/chung/Documents/MATLAB/SuiteSparse/CCOLAMD/Lib \
  -L/home/chung/Documents/MATLAB/SuiteSparse/CAMD/Lib -Wl,-rpath,. \
  -lcholmod -lamd -lcolamd -lsuitesparseconfig -lccolamd -lcamd \
  -llapacke -lblas -llapack -lopenblas -lm -lrt
